def self.parse_table(table)
  res = []
  table.each_line do |line|
    line.chomp!
    next unless line =~ /\s/
    res.push(line.gsub(",", "").split(/\s+/).map(&:to_i))
  end
  return res
end

def compute_tax_table(income, status)

  if year != 2020
    raise 'Tax table must be updated'
  end

  TAX_TABLE.each do |row|
    if income >= row[0] && income < row[1]
      return row[status.tax_table_column]
    end
  end
  raise "Income #{income} not found in tax table"
end
